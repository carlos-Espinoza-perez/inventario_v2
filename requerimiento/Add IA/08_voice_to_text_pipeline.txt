===========================================================================
🏗️ IMPLEMENTACIÓN DEL PIPELINE DE VOZ (DART)
===========================================================================

class VoiceAssistantService {
  final SpeechToText _speech = SpeechToText();
  bool _isAvailable = false;

  // 1. Inicializar el motor de audio
  Future<void> initSpeech() async {
    _isAvailable = await _speech.initialize(
      onError: (val) => print('Error de audio: $val'),
      onStatus: (val) => print('Estado: $val'),
    );
  }

  // 2. Escucha Activa
  Future<void> startListening({
    required Function(String) onResult,
    required Function() onDone,
  }) async {
    if (!_isAvailable) await initSpeech();

    await _speech.listen(
      onResult: (result) {
        if (result.finalResult) {
          onResult(result.recognizedWords);
          onDone();
        }
      },
      localeId: "es_ES", // Forzamos español para mejor precisión
      cancelOnError: true,
      listenMode: ListenMode.deviceDefault,
    );
  }

  void stop() => _speech.stop();
}
